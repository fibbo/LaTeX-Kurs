\section{Simple tikz commands}
Tikz stands for: \textit{Tikz ist kein Zeichenprogramm}, a recursive acronym. Tikz creates vector graphics which can be scaled arbitrarily.

A more extensive introduction to Tikz can be found here: \url{http://cremeronline.com/LaTeX/minimaltikz.pdf}


\subsection{Draw-Command}
\begin{tikzpicture}
\draw (0,0) --(1,2);
\end{tikzpicture}

\subsection{Optionen}

Draw a helper grid

\begin{tikzpicture}
\draw (0,0) --(1,2) -- (2,3) -- (1,0);
\draw[help lines, step=0.5] (0,0) grid (2,3);
\end{tikzpicture}

\begin{figure}[h]
\centering
\begin{tikzpicture}
\draw (0,0) -- (1,2) -- (2,3) -- (1,0);
\draw (0,3) -- (1.5,0.5);
\end{tikzpicture}
\caption{You can use Tikz figures in floating environments}
\end{figure}

\subsection{Vectors/Arrows}

\begin{tikzpicture}
\draw [<->] (0,2) -- (0,0) -- (3,0);
\draw [->] (0,0) -- (2,2);
\draw [<-] (0,1) -- (2,3);
\end{tikzpicture}

\subsection{Changing of line width}

\begin{tikzpicture}
\draw [ultra thick] (0,1) -- (2,1);
\draw [thick] (0,0.5) -- (2,0.5);
\draw [thin] (0,0) -- (2,0);
\end{tikzpicture}

Possible line widths:

\begin{itemize}
\item ultra thin
\item very thin
\item thin
\item semithick
\item thick
\item very thick
\item ultra thick
\end{itemize}

\section{Draw functions with Tikz}
\begin{center}
\begin{tikzpicture}[xscale=13,yscale=3.8] % You can also scale the drawing
\draw [<->] (0,0.8) -- (0,0) -- (0.5,0);
\draw[green, ultra thick, domain=0:0.5] plot (\x, {0.025+\x+\x*\x});
\end{tikzpicture}
\end{center}

\begin{figure}[h]
\centering
\begin{tikzpicture}[yscale=1.5] % Only y axis is scaled
\draw [help lines, <->] (0,0) -- (6.5,0);
\draw [help lines, ->] (0,-1.1) -- (0,1.1);
\draw [green,domain=0:2*pi] plot (\x, {(sin(\x r)* ln(\x+1))/2});
\draw [red,domain=0:pi] plot (\x, {sin(\x r)});
\draw [blue, domain=pi:2*pi] plot (\x, {cos(\x r)*exp(\x/exp(2*pi))});
\end{tikzpicture}
\caption{You can work with degrees or radian. If you want to work with radian you have to indicate this with \texttt{r}}
\end{figure}
